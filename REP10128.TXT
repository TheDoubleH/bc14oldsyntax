    [External]
    PROCEDURE ProcessVendorInvoices@1020000(VendorNo@1000 : Code[20];PeriodDate@1020000 : ARRAY [2] OF Date);
    VAR
      EntryApplicationManagement@1020001 : Codeunit 10202;
    BEGIN
      // search for invoices paid off by this payment
      EntryApplicationManagement.GetAppliedVendorEntries(TempVendorLedgerEntry,VendorNo,PeriodDate,TRUE);
      WITH TempVendorLedgerEntry DO BEGIN
        // search for invoices with 1099 amounts
        SETFILTER("Document Type",'%1|%2',"Document Type"::Invoice,"Document Type"::"Credit Memo");
        SETFILTER("IRS 1099 Amount",'<>0');
        SETRANGE("IRS 1099 Code",'MISC-','MISC-99');
        IF FINDSET THEN
          REPEAT
            IRS1099Management.Calculate1099Amount(
              Invoice1099Amount,Amounts,Codes,LastLineNo,TempVendorLedgerEntry,"Amount to Apply");
          UNTIL NEXT = 0;
      END;
    END;

    [External]
    PROCEDURE UpdateLines@3(Code@1000 : Code[10];Amount@1001 : Decimal) : Integer;
    BEGIN
      i := 1;
      WHILE (Codes[i] <> Code) AND (i <= LastLineNo) DO
        i := i + 1;

      IF (Codes[i] = Code) AND (i <= LastLineNo) THEN
        Amounts[i] := Amounts[i] + Amount
      ELSE
        ERROR(UnknownMiscCodeErr,
          TempVendorLedgerEntry."Entry No.",TempVendorLedgerEntry."Vendor No.",Code);
      EXIT(i);
    END;

    BEGIN
    END.
